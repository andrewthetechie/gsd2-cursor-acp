type AssistantMessageEvent =

  | { type: "text_end"; contentIndex: number; content: string; partial: AssistantMessage }
  | { type: "thinking_start"; contentIndex: number; partial: AssistantMessage }
  | { type: "thinking_delta"; contentIndex: number; delta: string; partial: AssistantMessage }
  | { type: "thinking_end"; contentIndex: number; content: string; partial: AssistantMessage }
  | { type: "toolcall_start"; contentIndex: number; partial: AssistantMessage }
  | { type: "toolcall_delta"; contentIndex: number; delta: string; partial: AssistantMessage }
  | { type: "toolcall_end"; contentIndex: number; toolCall: ToolCall; partial: AssistantMessage; malformedArguments?: boolean }

interface AssistantMessage {
  role: "assistant";
  content: (TextContent | ThinkingContent | ToolCall | ...)[]; // GSD-2 ToolCall, not ACP ToolCall

  model: string;
  usage: Usage;
  stopReason: StopReason;
  timestamp: number;
}

  input: number; output: number; cacheRead: number; cacheWrite: number; totalTokens: number;
  cost: { input: number; output: number; cacheRead: number; cacheWrite: number; total: number; };
}

// GSD-2 ToolCall (NOT ACP ToolCall):
interface ToolCall {  // from GSD-2 types.ts
  type: "toolCall";
  id: string;
  name: string;
  arguments: Record<string, any>;
}

interface TextContent { type: "text"; text: string; }
interface ThinkingContent { type: "thinking"; thinking: string; }

    Private method: `handleTextChunk(text: string): void`
    - If `activeBlockType !== 'text'`: close active block (call closeActiveBlock()), increment contentIndex, set activeBlockType = 'text', reset blockText = '', push TextContent to partial.content, push text_start event
    - Always: append text to blockText, update partial.content[contentIndex] text field, push text_delta event

    Private method: `handleThinkingChunk(text: string): void`
    - If `activeBlockType !== 'thinking'`: close active block, increment contentIndex, set activeBlockType = 'thinking', reset blockText = '', push ThinkingContent to partial.content, push thinking_start event
    - Always: append text to blockText, update partial.content[contentIndex] thinking field, push thinking_delta event

    Private method: `handleToolCallStart(toolCall: { toolCallId: string; title: string; status?: string }): void`
    - Close active block
    - Increment contentIndex
    - Set activeBlockType = 'toolcall'
    - Store activeToolCallId = toolCall.toolCallId, activeToolCallName = toolCall.title
    - Push a GSD-2 ToolCall to partial.content: `{ type: 'toolCall', id: toolCall.toolCallId, name: toolCall.title, arguments: {} }`
    - Push toolcall_start event

    Private method: `handleToolCallUpdate(update: { toolCallId: string; status?: string | null }): void`
    - If `update.status === 'completed' || update.status === 'failed'`:
      - Find the tool call in partial.content by id matching update.toolCallId
      - Push toolcall_end event: `{ type: 'toolcall_end', contentIndex: this.contentIndex, toolCall: foundToolCall, partial: this.partialSnapshot() }`
      - Reset activeBlockType = null (tool call block is closed)
    - Otherwise: ignore (intermediate updates not translated to events in Phase 3)

    Private method: `closeActiveBlock(): void`
    - If activeBlockType === 'text': push text_end event `{ type: 'text_end', contentIndex, content: blockText, partial: this.partialSnapshot() }`, set activeBlockType = null
    - If activeBlockType === 'thinking': push thinking_end event, set activeBlockType = null
    - If activeBlockType === 'toolcall': push toolcall_end with whatever partial ToolCall exists, set activeBlockType = null
    - If activeBlockType === null: no-op

    Public method: `handleDone(result: { stopReason?: string }): void`
    - If isDone: return (idempotent guard)
    - isDone = true
    - closeActiveBlock()
    - Map ACP stopReason to GSD-2 StopReason using:
      - 'end_turn' → 'stop'
      - 'max_tokens' → 'length'
      - 'max_turn_requests' → 'length'
      - 'cancelled' → 'stop'
      - default → 'stop'
    - Update partial.stopReason to mapped value
    - Build final message: `{ ...this.partial }` (copy)
    - Push done event: `{ type: 'done', reason: mappedReason as 'stop' | 'length', message: finalMessage }`
    - Call stream.end(finalMessage)

    Public method: `handleAbort(): void`
    - If isDone: return
    - isDone = true
    - closeActiveBlock()
    - Build error message: `{ ...this.partial, stopReason: 'aborted' as const }`
    - Push error event: `{ type: 'error', reason: 'aborted', error: errorMessage }`
    - Call stream.end(errorMessage)

    Private method: `partialSnapshot(): AssistantMessage`
    - Return `{ ...this.partial }` — a shallow copy to freeze the current state for the event

    IMPORTANT TypeScript note: The `partial` field on each event expects a valid `AssistantMessage`. Use `partialSnapshot()` for all event emissions so consumers don't observe mutation of the same object.
